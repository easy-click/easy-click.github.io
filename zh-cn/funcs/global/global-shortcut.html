3. 点击函数

3.1. clickPoint

  • 点击坐标
  • @param x x坐标
  • @param y y坐标
  • @return {boolean|布尔型}
loadECModule("global");
function main(){
    var result = clickPoint(100,100);
    if (result){
        toast("点击成功");
    } else {
        toast("点击失败");
    }
}
main();

4.11. isScrollEnd

  • 是否滚动到底部了，如果查不到元素也会返回false

loadECModule("global");

function main(){
    var selectors = S.get().className("android.widget.ListView");
    var result = isScrollEnd("UP",selectors);
    if (result){
        toast("滚动完成");
    } else {
        toast("滚动未完成");
    }
}
main();

9.4. shotNotification

  • 将通知发射处理，相当于点击了通知栏
  • @param seqId
  • @return {boolean|布尔型}
loadECModule("global");

function main(){
    //获取微信1条通知，然后进行点击
    var result = getLastNotification("com.tencent.mm",1);
    if (result!=null && result.length>0){
        var s = shotNotification(result[0].seqId);
        toast("结果:"+s);
    }   
}
main();
