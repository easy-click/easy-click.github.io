1. 说明

  • 代理事件模块运行的所有函数，是需要使用电脑激活才能使用的，免root
  • 代理事件模块的对象前缀是agentEvent，例如 agentEvent.clickPoint这样调用

loadECModule("agentEvent");

function main(){
    var selectors = S.get().className("android.widget.ListView");
    var result = agentEvent.isScrollEnd("UP",selectors);
    if (result){
        toast("滚动完成");
    }else {
        toast("滚动未完成");
    }
}
main();
